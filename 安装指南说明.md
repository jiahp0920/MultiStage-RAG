基础安装（最小化依赖）
# 使用 pip
pip install -r requirements.txt

# 或者使用 pip 直接安装核心依赖
pip install fastapi uvicorn pydantic chromadb dashscope redis sentence-transformers torch jieba

按需安装
# 安装特定向量存储
pip install chromadb               # 默认
pip install faiss-cpu              # FAISS
pip install pymilvus               # Milvus

# 安装特定重排序器
pip install dashscope              # 阿里百炼（默认）
pip install cohere                 # Cohere
pip install openai                 # OpenAI

# 安装开发依赖
pip install -e ".[dev]"

# 安装所有可选依赖
pip install -e ".[all]"

# 安装GPU支持
pip install -e ".[gpu]"

开发环境安装
# 克隆项目
git clone https://github.com/yourusername/MultiStage-RAG.git
cd MultiStage-RAG

# 创建虚拟环境
python -m venv venv
source venv/bin/activate  # Linux/Mac
# 或 venv\Scripts\activate  # Windows

# 安装开发版本
pip install -e ".[dev,test,all]"

# 设置预提交钩子
pre-commit install

Docker环境
FROM python:3.10-slim

WORKDIR /app

# 复制依赖文件
COPY requirements.txt .
COPY pyproject.toml .

# 安装依赖
RUN pip install --no-cache-dir -r requirements.txt

# 复制源代码
COPY src/ ./src/
COPY configs/ ./configs/

# 启动服务
CMD ["uvicorn", "src.multistage_rag.api.app:app", "--host", "0.0.0.0", "--port", "8000"]
